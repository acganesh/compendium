\begin{enumerate}
  \item Big $O$: upper bound.  Let $T(n), g(n)$ be functions of positive integers.  Formally, $T(n) = O(g(n))$ iff $\exists c, n_0 > 0$ such that $\forall n \geq n_0$, $0 \leq T(n) \leq c \cdot g(n)$.
  \item $\Omega(\dots)$ means a lower bound.  We say $T(n)$ is $\Omega(g(n))$ if $T(n)$ grows at least as fast as $g(n)$ as $n \to \infty$.  Formally, $T(n) = \Omega (g(n))$ iff $\exists c, n_0 > 0$ such that $\forall n \geq n_0$, we have $0 \leq c \cdot g(n) \leq T(n)$.
  \item We say $T(n) = \Theta (g(n))$ if both of the above are true.
\end{enumerate}
